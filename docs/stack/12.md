# Previous Smaller Element

Given an array `arr[]`, find the Previous Smaller Element (PSE) for every element in the array. The Previous Smaller Element of an element $x$ is defined as the first element to its left in the array that is smaller than $x$.

If no such element exists for a particular position, the PSE should be considered as $-1$.

## Example

```
 input = [ 1, 6, 2]
output = [-1, 1, 1]

 input = [ 1, 5,  0, 3, 4, 5]
output = [-1, 1, -1, 0, 3, 4]
```

## Solution

??? "Expand"

    Create a stack and process the elements one-by-one from left to right. For each element $e$ at index $i$:

    - remove all elements $x$ from the stack if $x \ge e$.
    - top of the stack will have the PSE of $e$.
    - add $e$ to the stack.

    ```kotlin
    fun findPrevSmaller(input: IntArray): IntArray {
      val stack = ArrayDeque<Int>()
      for ((i, n) in input.withIndex()) {
        while (stack.isNotEmpty() && stack.last() >= n) {
          stack.removeLast()
        }
        input[i] = if (stack.isEmpty()) -1 else stack.last()
        stack.add(n)
      }
      return input
    }
    ```

    package org.example

    import org.junit.jupiter.api.Assertions.assertArrayEquals
    import org.junit.jupiter.api.Test

    class PreviousSmallerElementTest {
      

      @Test
      fun `test example 1`() {
        val input = intArrayOf(1, 6, 2)
        val expected = intArrayOf(-1, 1, 1)
        assertArrayEquals(expected, findPrevSmaller(input))
      }

      @Test
      fun `test example 2`() {
        val input = intArrayOf(1, 5, 0, 3, 4, 5)
        val expected = intArrayOf(-1, 1, -1, 0, 3, 4)
        assertArrayEquals(expected, findPrevSmaller(input))
      }

      @Test
      fun `test strictly increasing array`() {
        val input = intArrayOf(10, 20, 30, 40)
        val expected = intArrayOf(-1, 10, 20, 30)
        assertArrayEquals(expected, findPrevSmaller(input))
      }

      @Test
      fun `test strictly decreasing array`() {
        // Every element is smaller than the previous one, so no smaller element exists to the left
        val input = intArrayOf(40, 30, 20, 10)
        val expected = intArrayOf(-1, -1, -1, -1)
        assertArrayEquals(expected, findPrevSmaller(input))
      }

      @Test
      fun `test array with identical elements`() {
        // The problem defines "smaller", not "smaller or equal"
        val input = intArrayOf(5, 5, 5, 5)
        val expected = intArrayOf(-1, -1, -1, -1)
        assertArrayEquals(expected, findPrevSmaller(input))
      }

      @Test
      fun `test single element array`() {
        val input = intArrayOf(100)
        val expected = intArrayOf(-1)
        assertArrayEquals(expected, findPrevSmaller(input))
      }

      @Test
      fun `test empty array`() {
        val input = intArrayOf()
        val expected = intArrayOf()
        assertArrayEquals(expected, findPrevSmaller(input))
      }

      @Test
      fun `test array with duplicates and peaks`() {
        val input = intArrayOf(1, 3, 0, 2, 5, 2)
        val expected = intArrayOf(-1, 1, -1, 0, 2, 0)
        assertArrayEquals(expected, findPrevSmaller(input))
      }
    }
