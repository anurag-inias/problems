## Solution

??? "Expand"

    We utilize the `insertAtBottom` as a subroutine here.

    ```kotlin
    fun <T> reverse(stack: ArrayDeque<T>) {
      if (stack.size < 2) return

      val top = stack.removeLast()
      reverse(stack)
      insertAtBottom(stack, top)
    }
    ```

    See the following example:

    ```
    reverse([1, 2, 3])
    - top = 3
    - [1, 2] => [2, 1]
    - 3 + [2, 1]
    ```

## Unit tests

??? "Expand"

    ```kotlin
    package org.example

    import kotlin.collections.ArrayDeque
    import org.junit.jupiter.api.Assertions.assertEquals
    import org.junit.jupiter.api.Assertions.assertTrue
    import org.junit.jupiter.api.Test

    class StackReverseTest {

      // Target signature: fun <T> reverse(stack: ArrayDeque<T>)

      @Test
      fun `should handle an empty stack`() {
        val stack = ArrayDeque<Int>()

        reverse(stack)

        assertTrue(stack.isEmpty(), "Stack should remain empty")
      }

      @Test
      fun `should handle a stack with one element`() {
        val stack = ArrayDeque<Int>()
        stack.addLast(1)

        reverse(stack)

        assertEquals(1, stack.size)
        assertEquals(1, stack.last(), "Single element should remain the same")
      }

      @Test
      fun `should reverse order of multiple elements`() {
        val stack = ArrayDeque<Int>()
        // Building the stack: Bottom [1, 2, 3] Top
        stack.addLast(1)
        stack.addLast(2)
        stack.addLast(3)

        reverse(stack)

        // After reversal, top should be 1
        assertEquals(3, stack.size)
        assertEquals(1, stack.removeLast())
        assertEquals(2, stack.removeLast())
        assertEquals(3, stack.removeLast())
      }

      @Test
      fun `should work with strings`() {
        val stack = ArrayDeque<String>()
        stack.addLast("A")
        stack.addLast("B")

        reverse(stack)

        // Old bottom "A" should now be the last element (top)
        assertEquals("A", stack.last())
        assertEquals(2, stack.size)
      }

      @Test
      fun `should maintain all elements after reversal`() {
        val stack = ArrayDeque<Int>()
        val range = 1..50
        range.forEach { stack.addLast(it) }

        reverse(stack)

        assertEquals(50, stack.size)
        assertEquals(1, stack.last(), "The very first item added should now be at the end (top)")
        assertEquals(50, stack.first(), "The very last item added should now be at the beginning (bottom)")
      }
    }
    ```
