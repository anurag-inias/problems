# Evaluate Postfix Expression

Given a string representing a postfix expression, evaluate it. Operators and operands are separated with whitespaces.

## Solution

??? "Expand"

    - Create a stack of int.
    - Process the expression token by token:
        - if an operand is encountered, put it in stack.
        - if an operator is encountered:
              - pop two operands from the stack
              - add back the result of applying the operator on these into stack.
    - stack now holds the value of the expression.

    ```kotlin
    import kotlin.math.pow

    fun evaluatePostfix(expression: String): Int {
      val operands = ArrayDeque<Int>()
      for (token in expression.split(" ")) {
        if (token.isBlank()) continue
        if (!token.isOperator) {
          operands.add(token.toInt())
          continue
        }
        val second = operands.removeLast()
        val first = operands.removeLast()
        operands.add(token.apply(first, second))
      }
      return operands.removeLast()
    }

    private fun String.apply(a: Int, b: Int): Int = when(this) {
      "+" -> a + b
      "-" -> a - b
      "*" -> a * b
      "/" -> a / b
      "^" -> a.toDouble().pow(b.toDouble()).toInt()
      else -> Int.MAX_VALUE
    }

    private val String.isOperator: Boolean
      get() = this in listOf("+", "-", "*", "/", "^")
    ```

## Unit tests

??? "Expand"

    ```kotlin
    package org.example

    import org.junit.jupiter.api.Assertions.assertEquals
    import org.junit.jupiter.api.Assertions.assertThrows
    import org.junit.jupiter.api.Test

    class PostfixEvaluationTest {

      // Method signature under test: fun evaluatePostfix(expression: String): Int

      @Test
      fun `should evaluate simple addition`() {
        val expression = "2 3 +"
        assertEquals(5, evaluatePostfix(expression))
      }

      @Test
      fun `should evaluate simple subtraction`() {
        val expression = "5 2 -"
        assertEquals(3, evaluatePostfix(expression))
      }

      @Test
      fun `should evaluate simple multiplication`() {
        val expression = "4 5 *"
        assertEquals(20, evaluatePostfix(expression))
      }

      @Test
      fun `should evaluate simple division`() {
        val expression = "10 2 /"
        assertEquals(5, evaluatePostfix(expression))
      }

      @Test
      fun `should evaluate complex expression with multiple operators`() {
        // (10 + 2) * 3 / (5 - 2) -> 12 * 3 / 3 -> 36 / 3 = 12
        // Postfix: "10 2 + 3 * 5 2 - /"
        val expression = "10 2 + 3 * 5 2 - /"
        assertEquals(12, evaluatePostfix(expression))
      }

      @Test
      fun `should handle expressions resulting in negative numbers`() {
        val expression = "2 5 -"
        assertEquals(-3, evaluatePostfix(expression))
      }

      @Test
      fun `should evaluate expression with multi-digit numbers`() {
        val expression = "100 200 +"
        assertEquals(300, evaluatePostfix(expression))
      }

      @Test
      fun `should return the value of a single operand expression`() {
        val expression = "42"
        assertEquals(42, evaluatePostfix(expression))
      }

      @Test
      fun `should throw exception for division by zero`() {
        val expression = "10 0 /"
        assertThrows(ArithmeticException::class.java) {
          evaluatePostfix(expression)
        }
      }

      @Test
      fun `should throw exception for invalid or malformed expression`() {
        // Too many operators
        val expression = "2 3 + *"
        assertThrows(Exception::class.java) {
          evaluatePostfix(expression)
        }
      }

      @Test
      fun `should handle expression with extra whitespace`() {
        val expression = "  10   5   +  "
        assertEquals(15, evaluatePostfix(expression))
      }
    }
    ```
