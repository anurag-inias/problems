Solution

Expand

\(O(n)\) push and \(O(1)\) pop.

Every time we enqueue an element, dequeue the elements preceding it and queue them behind new element.

class Stack<T> {

  private val queue = mutableListOf<T>()

  fun push(value: T) {
    queue.add(value)
    repeat(queue.size - 1) {
      queue.add(queue.removeFirst())
    }
  }

  fun peek(): T? = if (queue.isEmpty()) null else queue.first()

  fun pop(): T? = if (queue.isEmpty()) null else queue.removeFirst()

  fun size() = queue.size

  fun isEmpty() = queue.isEmpty()

}

package org.example

import org.assertj.core.api.Assertions.assertThat
import org.junit.jupiter.api.BeforeEach
import org.junit.jupiter.api.DisplayName
import org.junit.jupiter.api.Test

class SimpleStackTest {

  private lateinit var stack: Stack<String>

  @BeforeEach
  fun setup() {
    stack = Stack()
  }

  @Test
  @DisplayName("Should increase size and return element when pushing")
  fun `push should add element to stack`() {
    // When
    stack.push("Kotlin")

    // Then
    assertThat(stack.size()).isEqualTo(1)
    assertThat(stack.peek()).isEqualTo("Kotlin")
  }

  @Test
  @DisplayName("Should return elements in LIFO order when popping")
  fun `pop should return last pushed element (LIFO)`() {
    // Given
    stack.push("First")
    stack.push("Second")

    // When
    val popped = stack.pop()

    // Then
    assertThat(popped).isEqualTo("Second")
    assertThat(stack.size()).isEqualTo(1)
    assertThat(stack.peek()).isEqualTo("First")
  }

  @Test
  @DisplayName("Should return null when popping from empty stack")
  fun `pop on empty stack should return null`() {
    assertThat(stack.pop()).isNull()
  }

  @Test
  @DisplayName("Should verify stack is empty initially")
  fun `isEmpty should return true for new stack`() {
    assertThat(stack.isEmpty()).isTrue()

    stack.push("Item")
    assertThat(stack.isEmpty()).isFalse()
  }
}
